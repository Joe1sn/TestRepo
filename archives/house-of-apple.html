{

    if ((wint_t)_IO_WOVERFLOW (fp, WEOF) == WEOF)
      return EOF;
  if (_IO_in_backup (fp))

    4. 由于设置了p1->p2->_IO_write_ptr > p1->_IO_write_base，进入溢出处理p1->wide_vtable即p2

    5. 进入p2+0x18，本来是_IO_WOVERFLOW，被修改成了后门函数
